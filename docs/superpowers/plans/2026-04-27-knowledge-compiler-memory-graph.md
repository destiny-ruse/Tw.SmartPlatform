# Knowledge Compiler Memory Graph Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a repository-native Knowledge Compiler, memory graph, drift diagnostics, AI Skill layer, and Claude Code relative symlink adapter.

**Architecture:** The implementation mirrors the existing standards tooling pattern: Python standard-library CLI reads structured graph files, validates them, generates L0/L1/L2 indexes, reports Chinese diagnostics, and supports git-diff drift checks. AI Skills consume generated indexes and, when explicitly invoked, can maintain the formal graph while non-AI tooling remains the source of validation.

**Tech Stack:** Python standard library, `unittest`, JSON, a constrained YAML subset parser, Markdown Skill files, PowerShell-compatible commands.

---

Create `tools/knowledge/knowledge.py` with this content:

```python
#!/usr/bin/env python3
from __future__ import annotations

import argparse
import json
import re
import sys
from dataclasses import dataclass
from datetime import datetime, timezone
from pathlib import Path
from typing import Any


GENERATOR_VERSION = "1.0.0"
REPO_ROOT = Path(__file__).resolve().parents[2]
KNOWLEDGE_DIR = REPO_ROOT / "docs" / "knowledge"
GRAPH_DIR = KNOWLEDGE_DIR / "graph"
GENERATED_DIR = KNOWLEDGE_DIR / "generated"
TAXONOMY_PATH = KNOWLEDGE_DIR / "taxonomy.yaml"

REQUIRED_FIELDS = [
    "schema_version",
    "id",
    "kind",
    "name",
    "status",
    "summary",
    "owners",
    "tags",
    "source",
    "provenance",
]
ID_PATTERN = re.compile(r"^[a-z][a-z0-9-]*(\.[a-z0-9][a-z0-9-]*)+$")
VERSION_PATTERN = re.compile(r"^\d+\.\d+\.\d+$")
TOKEN_PATTERN = re.compile(r"^[a-z][a-z0-9-]*$")
GENERATED_SUFFIX = ".generated.json"


@dataclass(frozen=True)
class Diagnostic:
    level: str
    code: str
    location: str
    message_zh: str
    suggestion_zh: str = ""

    def format(self) -> str:
        prefix = "错误" if self.level == "ERROR" else "警告"
        result = f"{prefix} [{self.code}]\n位置: {self.location}\n说明: {self.message_zh}"
        if self.suggestion_zh:
            result += f"\n建议: {self.suggestion_zh}"
        return result

    def to_json(self) -> dict[str, str]:
        return {
            "severity": "error" if self.level == "ERROR" else "warning",
            "code": self.code,
            "path": self.location,
            "message_zh": self.message_zh,
            "suggestion_zh": self.suggestion_zh,
        }


@dataclass(frozen=True)
class GraphNode:
    path: Path
    data: dict[str, Any]
    lines: list[str]


def rel_path(path: Path) -> str:
    try:
        return path.resolve().relative_to(REPO_ROOT.resolve()).as_posix()
    except ValueError:
        return path.as_posix()


def error(code: str, path: Path | str, message_zh: str, suggestion_zh: str = "") -> Diagnostic:
    location = rel_path(path) if isinstance(path, Path) else path
    return Diagnostic("ERROR", code, location, message_zh, suggestion_zh)


def warn(code: str, path: Path | str, message_zh: str, suggestion_zh: str = "") -> Diagnostic:
    location = rel_path(path) if isinstance(path, Path) else path
    return Diagnostic("WARN", code, location, message_zh, suggestion_zh)


def has_errors(messages: list[Diagnostic]) -> bool:
    return any(message.level == "ERROR" for message in messages)


def emit(messages: list[Diagnostic]) -> None:
    for message in messages:
        print(message.format())


def read_text(path: Path) -> str:
    return path.read_text(encoding="utf-8")


def write_text(path: Path, content: str) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(content, encoding="utf-8", newline="\n")


def parse_scalar(raw_value: str) -> Any:
    value = raw_value.strip()
    if value == "":
        return None
    if value.startswith("[") and value.endswith("]"):
        inside = value[1:-1].strip()
        if inside == "":
            return []
        return [parse_scalar(item.strip()) for item in inside.split(",")]
    if (value.startswith('"') and value.endswith('"')) or (
        value.startswith("'") and value.endswith("'")
    ):
        return value[1:-1]
    if value == "true":
        return True
    if value == "false":
        return False
    return value


def parse_yaml_subset(lines: list[str]) -> dict[str, Any]:
    root: dict[str, Any] = {}
    stack: list[tuple[int, Any]] = [(-1, root)]
    pending_key: dict[int, tuple[dict[str, Any], str]] = {}

    for raw_line in lines:
        if raw_line.strip() == "" or raw_line.lstrip().startswith("#"):
            continue
        indent = len(raw_line) - len(raw_line.lstrip(" "))
        stripped = raw_line.strip()

        while stack and indent <= stack[-1][0]:
            stack.pop()
        parent = stack[-1][1]

        if stripped.startswith("- "):
            value_text = stripped[2:].strip()
            if not isinstance(parent, list):
                if indent not in pending_key:
                    raise ValueError(f"list item without list parent: {raw_line}")
                mapping, key = pending_key[indent]
                new_list: list[Any] = []
                mapping[key] = new_list
                stack.append((indent, new_list))
                parent = new_list
            if value_text.endswith(":"):
                item: dict[str, Any] = {value_text[:-1].strip(): {}}
                parent.append(item)
                stack.append((indent, item))
            elif ": " in value_text:
                key, value = value_text.split(": ", 1)
                parent.append({key.strip(): parse_scalar(value)})
            else:
                parent.append(parse_scalar(value_text))
            continue

        if ":" not in stripped:
            raise ValueError(f"unsupported yaml line: {raw_line}")

        key, value = stripped.split(":", 1)
        key = key.strip()
        value = value.strip()
        if not isinstance(parent, dict):
            raise ValueError(f"mapping entry under non-mapping parent: {raw_line}")
        if value == "":
            parent[key] = {}
            pending_key[indent + 2] = (parent, key)
            stack.append((indent, parent[key]))
        else:
            parent[key] = parse_scalar(value)

    return root


def graph_files() -> list[Path]:
    if not GRAPH_DIR.exists():
        return []
    return sorted(GRAPH_DIR.glob("*/*.yaml"))


def load_yaml_file(path: Path) -> tuple[dict[str, Any] | None, list[Diagnostic], list[str]]:
    lines = read_text(path).splitlines()
    try:
        return parse_yaml_subset(lines), [], lines
    except ValueError as exc:
        return None, [error("knowledge.yaml-parse", path, f"YAML 子集解析失败: {exc}")], lines


def load_taxonomy() -> tuple[dict[str, Any], list[Diagnostic]]:
    if not TAXONOMY_PATH.exists():
        return {}, [error("knowledge.taxonomy-missing", TAXONOMY_PATH, "缺少 taxonomy.yaml。")]
    data, messages, _ = load_yaml_file(TAXONOMY_PATH)
    if data is None:
        return {}, messages
    return data, messages


def load_graph_nodes() -> tuple[list[GraphNode], list[Diagnostic]]:
    nodes: list[GraphNode] = []
    messages: list[Diagnostic] = []
    for path in graph_files():
        data, file_messages, lines = load_yaml_file(path)
        messages.extend(file_messages)
        if data is not None:
            nodes.append(GraphNode(path=path, data=data, lines=lines))
    return nodes, messages


def collect_validation_messages() -> list[Diagnostic]:
    taxonomy, messages = load_taxonomy()
    valid_kinds = set(taxonomy.get("valid_kinds", []))
    valid_statuses = set(taxonomy.get("valid_statuses", []))
    nodes, graph_messages = load_graph_nodes()
    messages.extend(graph_messages)

    seen_ids: dict[str, Path] = {}
    for node in nodes:
        data = node.data
        for field in REQUIRED_FIELDS:
            if field not in data:
                messages.append(
                    error(
                        "knowledge.required-field",
                        node.path,
                        f"缺少必填字段 {field}。",
                        "补充该字段后重新运行 knowledge.py check。",
                    )
                )
        node_id = str(data.get("id", ""))
        if node_id and not ID_PATTERN.match(node_id):
            messages.append(error("knowledge.invalid-id", node.path, f"节点 ID {node_id} 不符合点分小写格式。"))
        if node_id in seen_ids:
            messages.append(error("knowledge.duplicate-id", node.path, f"节点 ID {node_id} 已在 {rel_path(seen_ids[node_id])} 中声明。"))
        elif node_id:
            seen_ids[node_id] = node.path
        schema_version = str(data.get("schema_version", ""))
        if schema_version and not VERSION_PATTERN.match(schema_version):
            messages.append(error("knowledge.invalid-version", node.path, f"schema_version {schema_version} 不符合语义化版本格式。"))
        kind = data.get("kind")
        if valid_kinds and kind and kind not in valid_kinds:
            messages.append(error("knowledge.invalid-kind", node.path, f"kind {kind} 不在 taxonomy.yaml 的 valid_kinds 中。"))
        status = data.get("status")
        if valid_statuses and status and status not in valid_statuses:
```

```python
def section_index(node: GraphNode) -> dict[str, Any]:
    sections: list[dict[str, Any]] = []
    for index, line in enumerate(node.lines, start=1):
        if ":" not in line or line.startswith(" "):
            continue
        key = line.split(":", 1)[0].strip()
        if not TOKEN_PATTERN.match(key.replace("_", "-")):
            continue
        start_line = index
        end_line = len(node.lines)
        for next_index in range(index + 1, len(node.lines) + 1):
            next_line = node.lines[next_index - 1]
            if next_line.strip() and not next_line.startswith(" ") and ":" in next_line:
                end_line = next_index - 1
                break
        sections.append({"key": key, "start_line": start_line, "end_line": end_line})
    return {
        "id": node.data["id"],
        "path": rel_path(node.path),
        "sections": sections,
    }


def build_edges(nodes: list[GraphNode]) -> list[dict[str, str]]:
    edges: list[dict[str, str]] = []

        node_id = str(data.get("id"))
        if data.get("kind") == "capability":
            modules = data.get("provided_by", {}).get("modules", [])
            for module_id in modules:
                if module_id in module_by_id:
                    edges.append({"from": module_id, "type": "provides", "to": node_id})
            for standard_id in data.get("standards", []):
                edges.append({"from": node_id, "type": "governed_by", "to": standard_id})
        if data.get("kind") == "module":
            capabilities = data.get("provides", {}).get("capabilities", [])
            for capability_id in capabilities:
                edges.append({"from": node_id, "type": "provides", "to": capability_id})
    unique = {(edge["from"], edge["type"], edge["to"]): edge for edge in edges}
    return [unique[key] for key in sorted(unique)]
```

```python
    lowered = text.lower()

    aliases = taxonomy.get("query_aliases", {})

            if str(phrase) in text and isinstance(mapped, list):
                tokens.update(str(item).lower() for item in mapped)
    return tokens
```
